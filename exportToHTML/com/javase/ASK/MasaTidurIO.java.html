125      //Memastikan pengguna memasukkan waktu Tidur dan waktu Bangun yang berformat BETUL
126      private static boolean isValidFormat(String value) {
127          Date date = null;
128          try {
129              SimpleDateFormat sdf = new SimpleDateFormat("HH:mm");
130              date = sdf.parse(value);
131              if (!value.equals(sdf.format(date))) {
132                  date = null;
133              }
134          } catch (ParseException ex) {
135              System.out.println(ex.getMessage());
136          }
137          return date != null;
138      }
